model_name = !cat slurm-86861.out | grep 'loading configuration file' | cut -d '/' -f 9 | head -1 
data_set = !cat  slurm-86861.out | grep 'Creating features from dataset file' | cut -d '/' -f 5 | head -1
loss = !cat  slurm-86861.out | grep  ' \'learning_rate\': ' | cut -d '{' -f 2 | sed s/}/''/g | cut -d ',' -f 1 | cut -d : -f 2
lr = !cat  slurm-86861.out | grep  ' \'learning_rate\': ' | cut -d '{' -f 2 | sed s/}/''/g | cut -d ',' -f 2 | cut -d : -f 2 
epoch = !cat  slurm-86861.out | grep  ' \'learning_rate\': ' | cut -d '{' -f 2 | sed s/}/''/g | cut -d ',' -f 3 | cut -d : -f 2
step = !cat  slurm-86861.out | grep  ' \'learning_rate\': ' | cut -d '{' -f 2 | sed s/}/''/g | cut -d ',' -f 4 | cut -d : -f 2
    
loss_array =[]
lr_array   = []
epoch_array = []
step_array  = []

for lo, l, e, s in zip(loss,lr,epoch,step):
    loss_array.append(float(lo.strip()))
    lr_array.append(float(l.strip()))
    epoch_array.append(float(e.strip()))
    step_array.append(float(s.strip()))
    
    
import matplotlib.pyplot as plt

plt.figure()
plt.plot(epoch_array,loss_array)
plt.title(model_name[0]+'_'+data_set[0])
plt.ylabel('Loss')
plt.xlabel('epoch')

plt.figure()
plt.plot(epoch_array,lr_array)
plt.title(model_name[0]+'_'+data_set[0])
plt.ylabel('Learning rate')
plt.xlabel('epoch')
